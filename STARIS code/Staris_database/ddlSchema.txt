-- Create People table
CREATE TABLE People (
    email TEXT PRIMARY KEY,
    name TEXT,
    department TEXT,
    job_title TEXT,
    engineering_pillars TEXT,
    research_keywords TEXT
);

-- Create Equipment table
CREATE TABLE Equipment (
    id INT AUTO_INCREMENT PRIMARY KEY,
    type TEXT,
    description TEXT,
    details TEXT,
    location TEXT,
    manager TEXT,
    access TEXT,
    contact TEXT,
    terms TEXT,
    cost TEXT,
    owner TEXT,
    keywords TEXT
);
